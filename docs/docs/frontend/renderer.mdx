# Renderer: renderer.js

The renderer handles all canvas drawing! It takes the game state and renders a frame!

## Colors

First, we define our color palette:

```javascript
const C = {
  bg:       '#080808',
  net:      'rgba(255,255,255,0.07)',
  left:     '#39ff14',
  leftDim:  'rgba(57,255,20,0.12)',
  right:    '#ff6bff',
  rightDim: 'rgba(255,107,255,0.12)',
  ball:     '#ffffff',
  ballGlow: 'rgba(255,255,255,0.2)',
}
```

## Renderer Class

Our renderer is a class:

```javascript
export class Renderer {
  constructor (canvas) {
    this.canvas = canvas
    this.ctx    = canvas.getContext('2d')
    this.frame  = 0
  }

  draw (state) {
    const { ctx } = this
    this.frame++

    ctx.fillStyle = C.bg
    ctx.fillRect(0, 0, W, H)

    ctx.save()
    ctx.strokeStyle = C.net
    ctx.lineWidth   = 2
    ctx.setLineDash([12, 10])
    ctx.beginPath()
    ctx.moveTo(W / 2, 0)
    ctx.lineTo(W / 2, H)
    ctx.stroke()
    ctx.setLineDash([])
    ctx.restore()

    for (const fx of (state.effects || [])) {
      const prog = fx.age / fx.maxAge
      ctx.beginPath()
      ctx.arc(fx.x, fx.y, 30 * prog, 0, Math.PI * 2)
      ctx.fillStyle = `rgba(255,255,255,${(1 - prog) * 0.4})`
      ctx.fill()
    }

    const glowR = BALL_R * 3.5
    const grd   = ctx.createRadialGradient(state.ball.x, state.ball.y, 0, state.ball.x, state.ball.y, glowR)
    grd.addColorStop(0, C.ballGlow)
    grd.addColorStop(1, 'transparent')
    ctx.fillStyle = grd
    ctx.beginPath()
    ctx.arc(state.ball.x, state.ball.y, glowR, 0, Math.PI * 2)
    ctx.fill()

    ctx.fillStyle = C.ball
    ctx.beginPath()
    ctx.arc(state.ball.x, state.ball.y, BALL_R, 0, Math.PI * 2)
    ctx.fill()

    this._paddle(ctx, LEFT_X,  state.left.y,  C.left,  C.leftDim)
    this._paddle(ctx, RIGHT_X, state.right.y, C.right, C.rightDim)

    if ((state.phase === 'countdown' || state.phase === 'point') && state.tick % 60 < 45) {
      const n = state.countdown > 0 ? state.countdown : ''
      if (n !== '') {
        ctx.save()
        ctx.font = 'bold 100px "Courier New", monospace'
        ctx.textAlign = 'center'
        ctx.textBaseline = 'middle'
        ctx.fillStyle = 'rgba(255,255,255,0.10)'
        ctx.fillText(String(n), W / 2, H / 2)
        ctx.restore()
      }
    }
  }

  _paddle (ctx, x, y, color, dimColor) {
    const grd = ctx.createLinearGradient(x - 8, y, x + PADDLE_W + 8, y)
    grd.addColorStop(0, dimColor)
    grd.addColorStop(0.5, dimColor)
    grd.addColorStop(1, 'transparent')
    ctx.fillStyle = grd
    ctx.beginPath()
    ctx.roundRect(x - 8, y, PADDLE_W + 16, PADDLE_H, 4)
    ctx.fill()

    ctx.fillStyle = color
    ctx.beginPath()
    ctx.roundRect(x, y, PADDLE_W, PADDLE_H, 4)
    ctx.fill()
  }

  drawWaiting () {
    const { ctx } = this
    ctx.fillStyle = C.bg
    ctx.fillRect(0, 0, W, H)
    ctx.font = '14px "Courier New", monospace'
    ctx.textAlign = 'center'
    ctx.textBaseline = 'middle'
    ctx.fillStyle = 'rgba(255,255,255,0.18)'
    ctx.fillText('awaiting connection…', W / 2, H / 2)
    ctx.textAlign = 'left'
  }
}
```
